# GitHub Secrets Required for CI/CD Pipeline
# Go to: GitHub Repo -> Settings -> Secrets and variables -> Actions -> New repository secret
# Add each of the following:

AWS_ACCESS_KEY_ID=<your-aws-access-key>
AWS_SECRET_ACCESS_KEY=<your-aws-secret-key>
EC2_HOST=<your-ec2-public-ip-or-dns>
EC2_USER=ec2-user
EC2_KEY=<your-ec2-private-key-pem-content>
RDS_PASSWORD=<your-rds-database-password>
JWT_SECRET=<your-jwt-secret-min-256-bits>
S3_BUCKET=<your-s3-bucket-name>
CLOUDFRONT_DISTRIBUTION_ID=<your-cloudfront-distribution-id>

# NEVER commit real values to the repository!
